// src/services/uploadService.ts // Complete video upload pipeline — no AI engine dependency whatsoever import * as FileSystem from 'expo-file-system' import { api } from '../lib/api' import { supabase } from '../lib/supabase' import { useStore } from '../lib/store' export interface UploadProgress { stage: 'preparing' | 'uploading' | 'processing' | 'done' | 'error' percent: number videoId?: string error?: string } export interface UploadRequest { videoUri: string sport: string technique: string durationSec: number is1v1?: boolean partnerId?: string onProgress?: (p: UploadProgress) => void } export async function uploadVideo(req: UploadRequest): Promise<string> { const { videoUri, sport, technique, durationSec, is1v1, partnerId, onProgress } = req const notify = (p: UploadProgress) => onProgress?.(p) // Step 1: Get signed upload URL from our API notify({ stage: 'preparing', percent: 5 }) const { data: urlData } = await api.post('/video/upload-url', { sport, technique, duration_sec: durationSec, is_1v1: is1v1 ?? false, fighter_partner_id: partnerId ?? null }) const { upload_url, video_id, content_type } = urlData // Step 2: Upload directly to Cloudflare R2 via signed URL // expo-file-system handles large files without loading into memory notify({ stage: 'uploading', percent: 10 }) const uploadResult = await FileSystem.uploadAsync(upload_url, videoUri, { httpMethod: 'PUT', uploadType: FileSystem.FileSystemUploadType.BINARY_CONTENT, headers: { 'Content-Type': content_type ?? 'video/mp4' }, sessionType: FileSystem.FileSystemSessionType.BACKGROUND, }) if (uploadResult.status !== 200 && uploadResult.status !== 201) { throw new Error(`Upload failed: ${uploadResult.status}`) } notify({ stage: 'uploading', percent: 75 }) // Step 3: Tell API the upload is complete — triggers moderation + analysis queue await api.post(`/video/${video_id}/complete`) notify({ stage: 'processing', percent: 90, videoId: video_id }) // Step 4: Poll for analysis completion (returns fast for mock, slower for real) const analysisId = await pollForAnalysis(video_id, onProgress) notify({ stage: 'done', percent: 100, videoId: analysisId }) return analysisId } async function pollForAnalysis( videoId: string, onProgress?: (p: UploadProgress) => void ): Promise<string> { const MAX_POLLS = 24 // 2 minutes maximum const POLL_INTERVAL = 5000 for (let i = 0; i < MAX_POLLS; i++) { await sleep(POLL_INTERVAL) const { data } = await api.get(`/video/${videoId}/status`) // Animate progress while waiting (80% → 99%) const progress = Math.min(99, 80 + (i / MAX_POLLS) * 19) onProgress?.({ stage: 'processing', percent: progress, videoId }) if (data.analysis_status === 'done') return data.analysis_id if (data.analysis_status === 'failed') throw new Error('Analysis failed') } throw new Error('Analysis timed out') } const sleep = (ms: number) => new Promise(r => setTimeout(r, ms))

// sportsai-api/src/services/xpService.ts — Complete XP + belt + unlock system import { supabase } from '../lib/supabase' import { sendPush } from './pushService' import { checkUnlocks } from './unlockService' export type XPEvent = | 'upload' | 'score_improved_10pct' | 'score_improved_20pct' | 'coach_review_completed' | 'streak_week' | 'challenge_completed' | 'technique_unlocked' | 'belt_promoted' | 'first_1v1_win' | 'perfect_score' const XP_VALUES: Record<XPEvent, number> = { upload: 50, score_improved_10pct: 120, score_improved_20pct: 200, coach_review_completed: 200, streak_week: 75, challenge_completed: 150, technique_unlocked: 300, belt_promoted: 500, first_1v1_win: 250, perfect_score: 400 } const BELT_THRESHOLDS = [ { name: 'white', xp: 0 }, { name: 'yellow', xp: 500 }, { name: 'orange', xp: 1200 }, { name: 'green', xp: 2500 }, { name: 'blue', xp: 5000 }, { name: 'purple', xp: 9000 }, { name: 'brown', xp: 15000 }, { name: 'black', xp: 25000 } ] export interface XPResult { xp_awarded: number total_xp: number weekly_xp: number belt_promoted: boolean new_belt?: string prev_belt?: string techniques_unlocked: string[] } export async function awardXP( athleteId: string, event: XPEvent, sourceId: string, metadata?: { sport?: string; technique?: string; score?: number } ): Promise<XPResult> { const amount = XP_VALUES[event] // Get current XP state const { data: current } = await supabase .from('athlete_xp').select('*').eq('athlete_id', athleteId).single() const newTotal = (current?.total_xp ?? 0) + amount const newWeekly = (current?.weekly_xp ?? 0) + amount // Calculate new belt const newBelt = BELT_THRESHOLDS.filter(b => newTotal >= b.xp).pop()?.name ?? 'white' const beltPromoted = newBelt !== current?.current_belt // Calculate belt progress percentage const currentBeltIdx = BELT_THRESHOLDS.findIndex(b => b.name === newBelt) const nextBelt = BELT_THRESHOLDS[currentBeltIdx + 1] const beltProgress = nextBelt ? Math.round(((newTotal - BELT_THRESHOLDS[currentBeltIdx].xp) / (nextBelt.xp - BELT_THRESHOLDS[currentBeltIdx].xp)) * 100) : 100 // Update XP in DB (upsert) await supabase.from('athlete_xp').upsert({ athlete_id: athleteId, total_xp: newTotal, weekly_xp: newWeekly, current_belt: newBelt, belt_progress_pct: beltProgress }) // Log transaction await supabase.from('xp_transactions').insert({ athlete_id: athleteId, amount, reason: event, source_id: sourceId }) // Check technique unlocks const unlocked: string[] = [] if (metadata?.sport && metadata?.technique && metadata?.score) { const newUnlocks = await checkUnlocks(athleteId, metadata.sport, metadata.technique, metadata.score) unlocked.push(...newUnlocks) } // Send push notification for belt promotion if (beltPromoted) { await sendPush(athleteId, { title: `🏅 Belt Promotion!`, body: `You've been promoted to ${newBelt.charAt(0).toUpperCase() + newBelt.slice(1)} Belt!`, data: { type: 'belt_promotion', new_belt: newBelt } }) } // Send push for technique unlocks for (const tech of unlocked) { await sendPush(athleteId, { title: `🔓 Technique Unlocked!`, body: `You've unlocked: ${tech.replace(/_/g, ' ')}!`, data: { type: 'technique_unlock', technique: tech } }) } return { xp_awarded: amount, total_xp: newTotal, weekly_xp: newWeekly, belt_promoted: beltPromoted, new_belt: beltPromoted ? newBelt : undefined, prev_belt: beltPromoted ? current?.current_belt : undefined, techniques_unlocked: unlocked } }
