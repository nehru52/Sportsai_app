Step 2 — Collect pro/Olympic footage (legally)

Legal method — extract skeletons, not frames

The Copyright-Safe Pro Data Strategy

You cannot store copyrighted broadcast frames. But pose coordinates extracted from those frames are not copyrightable — they are mathematical measurements of human body position. This is how every research lab does it.

✓ Legal Store: JSON coordinates only Delete: source video after extraction 
# skeleton_extractor.py — run on any video, store only the data
import cv2, json
from ultralytics import YOLO

model = YOLO('yolo11x-pose.pt')  # fine-tuned checkpoint

def extract_skeleton_sequence(video_path, athlete_id, technique_label, skill_level):
    cap = cv2.VideoCapture(video_path)
    frames = []
    frame_idx = 0
    
    while cap.isOpened():
        ret, frame = cap.read()
        if not ret: break
        
        results = model(frame, verbose=False)
        
        if results[0].keypoints is not None:
            kpts = results[0].keypoints.xyn[0].tolist()
            conf = results[0].keypoints.conf[0].tolist()
            
            frames.append({
                "frame": frame_idx,
                "timestamp_ms": cap.get(cv2.CAP_PROP_POS_MSEC),
                "keypoints": kpts,      # normalized x,y per joint
                "confidence": conf,    # per-joint confidence
                "fps": cap.get(cv2.CAP_PROP_FPS)
            })
        frame_idx += 1
    
    cap.release()
    
    # Store only the skeleton data — not the video
    return {
        "athlete_id": athlete_id,
        "technique": technique_label,
        "skill_level": skill_level,  # "elite", "advanced", "intermediate", "beginner"
        "frames": frames,
        "source": "extracted"     # never store original video path
    }

def extract_biomechanics(results, frame_data):
    kpts = results[0].keypoints.xyn[0].numpy()
    return {
        "timestamp": frame_data['ts'],
        "joints": {
            "hip_angle": calc_angle(kpts[11], kpts[23], kpts[25]),
            "knee_angle": calc_angle(kpts[23], kpts[25], kpts[27]),
            "spine_angle": calc_angle(kpts[5], kpts[11], kpts[23]),
            "head_height_norm": kpts[0][1],  # normalized y
            "hip_height_norm": kpts[23][1],
            "shoulder_width": abs(kpts[5][0] - kpts[6][0]),
        },
        "velocity": calc_velocity(kpts, frame_data['prev_kpts'])
    }

Layer 3 — Rules Engine (your moat, build this first)

This is the single most important file in your codebase. Every threshold comes from your coaches and your elite athlete data. This is not ML — it is structured biomechanical knowledge.

Phase 1 vs Phase 3 architecture — read this: This threshold-based rules engine is intentional scaffolding for Phase 1. A peer review correctly noted that hard-coded thresholds eventually fail across different body types (tall vs. stocky wrestlers) and as technique evolves across eras. The correct long-term architecture is a learned distribution model trained on 1,000+ elite sequences. But you cannot build that model without the data — and you don't have the data yet. The migration path: (1) Ship this threshold engine now. (2) Collect 500+ labeled sequences over Phase 1–2 using the coach divergence tracking table. (3) Replace Layer 3 with the AdaptiveRulesEngine in Phase 2 once you have the training data. The two are sequential, not competing. 
# rules_engine/wrestling.py
# Every threshold validated against AthletePose3D elite data + coach review

WRESTLING_RULES = {
    "single_leg_shot": {
        "entry_phase": {
            "level_change": {
                "metric": "hip_height_norm",
                "elite_range": [0.55, 0.70],   # from pro data extraction
                "error_if_above": 0.70,
                "severity": "high",
                "cue": "Drop your level — hips should be near knee height on the step-in",
                "visual_timestamp": "entry"
            },
            "head_position": {
                "metric": "head_angle",
                "elite_range": [160, 180],
                "error_if_below": 150,
                "severity": "medium",
                "cue": "Head up — eyes on the opponent's chest",
                "visual_timestamp": "entry"
            },
        },
        "execution_phase": {
            "knee_drive": {
                "metric": "knee_velocity_forward",
                "elite_range": [0.8, 2.5],   # normalized velocity units
                "error_if_below": 0.8,
                "severity": "high",
                "cue": "Drive through — your penetration step needs more explosive force",
                "visual_timestamp": "contact"
            },
        }
    },
    "double_leg": { ... },
    "sprawl": { ... }
}

def assess_technique(pose_sequence, technique_name):
    rules = WRESTLING_RULES[technique_name]
    errors = []
    score = 100
    
    phases = detect_technique_phases(pose_sequence)
    
    for phase_name, phase_rules in rules.items():
        phase_frames = phases.get(phase_name, [])
        if not phase_frames: continue
        
        avg_metrics = average_phase_metrics(phase_frames)
        
        for rule_name, rule in phase_rules.items():
            value = avg_metrics.get(rule['metric'])
            if value is None: continue
            
            error_triggered = False
            if 'error_if_above' in rule and value > rule['error_if_above']:
                error_triggered = True
            if 'error_if_below' in rule and value < rule['error_if_below']:
                error_triggered = True
            
            if error_triggered:
                severity_deduction = {"high": 20, "medium": 10, "low": 5}
                score -= severity_deduction[rule['severity']]
                errors.append({
                    "phase": phase_name,
                    "rule": rule_name,
                    "severity": rule['severity'],
                    "measured_value": round(value, 2),
                    "elite_range": rule['elite_range'],
                    "cue": rule['cue'],
                    "jump_to_timestamp": rule['visual_timestamp']
                })
    
    return {"score": max(0, score), "errors": errors, "technique": technique_name}

Layer 3b — Adaptive Rules Engine (Phase 2 migration target)

Once you have 500+ labeled sequences from the coach divergence loop, migrate Layer 3 to this architecture. It learns the elite distribution from data instead of relying on fixed thresholds — handling body-type variation and technique evolution automatically.

# rules_engine/adaptive.py — Phase 2 replacement for threshold engine
# Requirements: 500+ labeled elite sequences per technique before training

from sklearn.mixture import GaussianMixture
from sklearn.ensemble import IsolationForest
import joblib, numpy as np

class AdaptiveRulesEngine:
    """
    Replaces hard-coded thresholds with a learned elite distribution.
    Train when you have 500+ labeled elite sequences per technique.
    Until then, use the threshold-based rules_engine/wrestling.py.
    """
    def __init__(self, sport: str, technique: str):
        self.sport = sport
        self.technique = technique
        # Gaussian Mixture Model fitted on elite athlete sequences
        self.elite_gmm = joblib.load(f'models/{sport}_{technique}_gmm.pkl')
        # Isolation Forest for anomaly detection (catches novel errors)
        self.anomaly_detector = joblib.load(f'models/{sport}_{technique}_iso.pkl')

    def assess(self, pose_sequence):
        features = self._extract_features(pose_sequence)

        # Similarity to elite distribution (replaces hard threshold)
        log_likelihood = self.elite_gmm.score_samples([features])[0]
        similarity_score = self._normalize_score(log_likelihood)

        # Detect anomalous movement patterns (catches errors your rules missed)
        anomaly_score = self.anomaly_detector.decision_function([features])[0]
        errors = self._classify_errors(features, anomaly_score)

        return {
            'score': similarity_score,
            'percentile': self._percentile(similarity_score),
            'errors': errors,
            'confidence': min(1.0, len(self.training_sequences) / 1000)
        }

# Training script — run once you hit 500+ labeled sequences
def train_adaptive_engine(sport, technique, labeled_sequences):
    features = [extract_features(seq) for seq in labeled_sequences
                if seq['skill_level'] == 'elite']
    
    gmm = GaussianMixture(n_components=5, covariance_type='full')
    gmm.fit(features)
    joblib.dump(gmm, f'models/{sport}_{technique}_gmm.pkl')
    
    iso = IsolationForest(contamination=0.1, random_state=42)
    iso.fit(features)
    joblib.dump(iso, f'models/{sport}_{technique}_iso.pkl')
    print(f"Trained on {len(features)} elite sequences. Ready to replace threshold engine.")

04b — Phase 1 / Annotation Player

The Annotation Player

This is the most important screen in the entire product. A peer review correctly flagged it as underspecified. Everything else is plumbing — this screen is where trust is built or lost. Treat it as its own component with its own spec.

Why this screen determines product success: Every sports AI product that has failed did so because the analysis output was a wall of numbers or a vague text report. The moment an athlete taps an error and sees exactly what went wrong — at that exact moment — is when they believe the AI. Before that tap, it's a demo. After it, it's a product they tell their teammates about.

The 5 interactions — all required, not optional

Interaction 1 — The most important one

Tap error → jump to 2 seconds BEFORE the error

Not to the error frame. 2 seconds before it. Athletes need to see the setup that caused the mistake — not just the mistake. "Your hip was too high at 0:04" is useless if the video starts at 0:04. Jump to 0:02 and let them watch it unfold. This is a practitioner-level insight that separates coaching software from analytics software.

Interaction 2

Automatic slow-motion at error moment

When the video reaches the flagged timestamp, drop to 0.25x speed automatically. Resume normal speed after 2 seconds. The athlete sees the error in detail without having to manually scrub. Implement with Cloudflare Stream's playback rate API.

Interaction 3 — The viral feature

Elite comparison toggle overlay

One button: "Compare to elite." The screen splits or overlays the athlete's skeleton with an elite athlete performing the same technique, synced to the same phase (entry/execution/finish — not absolute time). The gap is instantly visible. This is the feature athletes screenshot and post. Build the share sheet on day one.

Interaction 4

Swipe left/right between errors

Each swipe jumps to the next flagged error, pre-loaded and ready. No scrubbing. No list-tapping. Just swipe through your mistakes in sequence. Athletes should be able to review all 3 errors in under 30 seconds.

Interaction 5

Pinch to zoom on body region

Athlete pinches to zoom into the knee, hip, or head area during the error frame. The skeleton overlay stays anchored. This is especially important for fine-grained errors like head position or hand placement that are hard to see at full-frame zoom.

// AnnotationPlayer.tsx — the most important component you will build
import { Video, ResizeMode } from 'expo-av'
import { Gesture, GestureDetector } from 'react-native-gesture-handler'
import Animated from 'react-native-reanimated'
import { Canvas, Path, Circle } from '@shopify/react-native-skia'

interface AnnotationPlayerProps {
  videoUrl: string               // Cloudflare Stream HLS URL
  poseOverlay: PoseSequence      // 30fps skeleton JSON from R2
  errorMarkers: ErrorMarker[]    // [{timestamp_ms, rule, severity, cue}]
  eliteComparison?: PoseSequence // elite skeleton for same technique
}

const AnnotationPlayer = ({ videoUrl, poseOverlay, errorMarkers, eliteComparison }) => {
  const videoRef = useRef(null)
  const [currentError, setCurrentError] = useState(0)
  const [showElite, setShowElite] = useState(false)
  const [currentPose, setCurrentPose] = useState(null)

  // Jump to 2s BEFORE the error timestamp
  const jumpToError = async (errorIdx) => {
    const error = errorMarkers[errorIdx]
    const jumpMs = Math.max(0, error.timestamp_ms - 2000) // 2s before
    await videoRef.current.setPositionAsync(jumpMs)
    await videoRef.current.setRateAsync(1.0, true)

    // Auto slow-mo when reaching the error frame
    const slowMoTimer = setTimeout(async () => {
      await videoRef.current.setRateAsync(0.25, true)
      setTimeout(async () => {
        await videoRef.current.setRateAsync(1.0, true)
      }, 2000)
    }, 2000) // fires when we reach the error timestamp
    setCurrentError(errorIdx)
  }

  // Swipe gesture: left = next error, right = previous
  const swipe = Gesture.Pan().onEnd((e) => {
    if (e.translationX < -50 && currentError < errorMarkers.length - 1)
      runOnJS(jumpToError)(currentError + 1)
    if (e.translationX > 50 && currentError > 0)
      runOnJS(jumpToError)(currentError - 1)
  })

  return (
    <GestureDetector gesture={swipe}>
      <View style={styles.container}>
        <Video ref={videoRef} source={{ uri: videoUrl }}
          style={styles.video} resizeMode={ResizeMode.CONTAIN} />

        {/* Skeleton overlay — rendered on Skia canvas */}
        <Canvas style={StyleSheet.absoluteFill}>
          {currentPose && <SkeletonOverlay pose={currentPose}
            highlight={errorMarkers[currentError]?.joints} />}
          {showElite && eliteComparison && <SkeletonOverlay
            pose={syncedElitePose} color="rgba(78,205,196,0.6)" />}
        </Canvas>

        {/* Error pill strip at bottom */}
        <ErrorStrip errors={errorMarkers} current={currentError}
          onTap={jumpToError} />

        {/* Elite toggle */}
        <TouchableOpacity onPress={() => setShowElite(!showElite)}
          style={[styles.eliteBtn, showElite && styles.eliteBtnActive]}>
          <Text>{showElite ? 'Hide elite' : 'Compare to elite'}</Text>
        </TouchableOpacity>
      </View>
    </GestureDetector>
  )
}

07 — Phase 2 / Coach Knowledge

Encoding Coach Knowledge at Scale

The structured process to turn what coaches know into what the AI knows. This is a 6-month ongoing program, not a one-time event.

The Coach Knowledge Interview Protocol

Run this with each coach you onboard. Record everything. This is where your rules engine thresholds come from.

Session 1 — Error Taxonomy (2 hours)

Ask: "For the single-leg shot, what are the 5 most common errors you see in beginners? In advanced athletes?" Record them categorizing clips from your dataset. You're building a vocabulary of errors — each one must eventually map to a measurable metric.

Session 2 — Biomechanical Translation (3 hours)

For each error from Session 1: "When you see this, what EXACTLY are you watching? Which body part? What angle? What timing? How early can you predict it will happen?" Push until every error has a joint + angle + timing attached. "Hips too high" → "hip height normalized above 0.70 at moment of step-in."

Session 3 — Elite Baseline Calibration (2 hours)

Show the coach your extracted elite athlete skeleton data. Ask them to rate each clip 1–10. Use their ratings to validate your threshold ranges. If they rate a clip 9/10 but your rules engine gives it 65/100, your thresholds are wrong — not the coach.

Ongoing — Coach Correction Loop

Every time a coach edits the AI's analysis (changes the score, adds an error the AI missed, removes a false positive), that correction is stored. After 100+ corrections per technique, you have a training dataset to improve the rules engine. This is how the AI gets smarter without you doing anything extra.

-- Find available coach for a review request
SELECT c.id, c.display_name, c.avg_rating, c.response_time_hrs,
       c.completed_reviews, c.sport_specialties
FROM users c
JOIN coach_availability ca ON c.id = ca.coach_id
WHERE c.role = 'coach'
  AND 'wrestling' = ANY(c.sport_specialties)
  AND ca.is_available = true
  AND c.id NOT IN (
    -- exclude coaches with 5+ pending reviews (overloaded)
    SELECT coach_id FROM coach_reviews 
    WHERE status = 'pending' 
    GROUP BY coach_id HAVING COUNT(*) >= 5
  )
ORDER BY (c.avg_rating * 0.4 + (1.0 / c.response_time_hrs) * 0.4 
  + (c.completed_reviews / 100.0) * 0.2) DESC
LIMIT 3;

  let lastFeedbackAt = 0

          }))
        }
        lastFeedbackAt = Date.now()
      }
    }
  })

11 — Phase 3 / Biomechanics Engine

Olympic-Grade Biomechanics Engine

Phase 1 rules engine uses simple thresholds. Phase 3 adds temporal modeling, multi-joint coordination analysis, and predictive injury scoring.

Upgrade 1 — Temporal sequence modeling

Simple threshold checks miss timing. An elite single-leg shot isn't just about each joint angle — it's about the exact sequence of how they move. Add an LSTM layer on top of your rules engine to learn technique "fingerprints" from your elite skeleton dataset.

# temporal_model.py — phase 3 upgrade
import torch
import torch.nn as nn

class TechniqueClassifier(nn.Module):
    """
    Takes a sequence of pose frames → classifies technique quality
    Trained on your elite skeleton dataset + coach-labeled quality scores
    """
    def __init__(self, input_size=66, hidden_size=256, num_classes=4):
        # input_size = 33 joints × 2 (x,y) coordinates
        # num_classes = elite / advanced / intermediate / beginner
        super().__init__()
        self.lstm = nn.LSTM(input_size, hidden_size, num_layers=2, 
                           batch_first=True, dropout=0.3)
        self.attention = nn.MultiheadAttention(hidden_size, num_heads=8)
        self.classifier = nn.Sequential(
            nn.Linear(hidden_size, 128),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(128, num_classes)
        )
    
    def forward(self, pose_sequence):
        # pose_sequence: [batch, frames, 66 keypoint values]
        lstm_out, _ = self.lstm(pose_sequence)
        attended, _ = self.attention(lstm_out, lstm_out, lstm_out)
        return self.classifier(attended[:, -1, :])  # last hidden state
